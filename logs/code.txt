async def run() -> dict:
    domain = "strique.io"
    steps = []

    # Step 1: Retrieve a list of current backlinks for strique.io.
    try:
        tool_result = await run_tool("backlinks_backlinks", {
            "target": domain,
            "mode": "as_is",
            "limit": 10,
            "offset": 0
        })
        raw_results = tool_result
        if isinstance(tool_result, list):
            key_insights = f"Retrieved {len(tool_result)} backlinks."
        else:
            key_insights = f"Retrieved result of type {type(tool_result).__name__}"
    except Exception as e:
        raw_results = {"error": str(e), "error_type": type(e).__name__}
        key_insights = f"Error: {str(e)}"
    
    steps.append({
        "step_id": 1,
        "description": "Retrieve a list of current backlinks for strique.io.",
        "raw_results": raw_results,
        "key_insights": key_insights,
    })

    # Step 2: Analyze the quality and relevance of the backlinks for strique.io.
    try:
        tool_result = await run_tool("backlinks_summary", {
            "target": domain,
            "include_subdomains": True,
            "exclude_internal_backlinks": True
        })
        raw_results = tool_result
        if isinstance(tool_result, dict):
            key_insights = f"Retrieved data with {len(tool_result)} keys."
        else:
            key_insights = f"Retrieved result of type {type(tool_result).__name__}"
    except Exception as e:
        raw_results = {"error": str(e), "error_type": type(e).__name__}
        key_insights = f"Error: {str(e)}"
    
    steps.append({
        "step_id": 2,
        "description": "Analyze the quality and relevance of the backlinks for strique.io.",
        "raw_results": raw_results,
        "key_insights": key_insights,
    })

    # Step 3: Identify potential opportunities for acquiring new high-quality backlinks.
    try:
        tool_result = await run_tool("backlinks_competitors", {
            "target": domain,
            "limit": 10,
            "offset": 0,
            "main_domain": True,
            "exclude_large_domains": True,
            "exclude_internal_backlinks": True
        })
        raw_results = tool_result
        if isinstance(tool_result, list):
            key_insights = f"Retrieved {len(tool_result)} competitor domains."
        else:
            key_insights = f"Retrieved result of type {type(tool_result).__name__}"
    except Exception as e:
        raw_results = {"error": str(e), "error_type": type(e).__name__}
        key_insights = f"Error: {str(e)}"
    
    steps.append({
        "step_id": 3,
        "description": "Identify potential opportunities for acquiring new high-quality backlinks.",
        "raw_results": raw_results,
        "key_insights": key_insights,
    })

    result = {
        "summary": "Backlinks analysis for strique.io completed with insights on current backlinks, their quality, and new opportunities.",
        "steps": steps
    }

    return result
